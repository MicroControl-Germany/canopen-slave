//--------------------------------------------------------------------------------------------------------------------//
// Cos404_AI_GetDefaultType() //
// //
//--------------------------------------------------------------------------------------------------------------------//
uint16_t Cos404_AI_GetDefaultType(uint8_t ubChannelV)
{
uint16_t uwTypeV;
//---------------------------------------------------------------------------------------------------
// Check valid AI channel
//
if (ubChannelV >= eCOS404_AI_CHANNEL_MAX)
{
return 0;
}
//---------------------------------------------------------------------------------------------------
//
uwTypeV = eCOS404_AI_ST_VOLT;
return uwTypeV;
}
//--------------------------------------------------------------------------------------------------------------------//
// Cos404_AI_Process() //
// //
//--------------------------------------------------------------------------------------------------------------------//
void Cos404_AI_Process(void)
{
uint8_t ubChannelT;
int32_t slFieldValueT;
int32_t slProcessValueT;
//---------------------------------------------------------------------------------------------------
// Provide new values to COS 404 AI module
//
ubChannelT = eCOS404_AI_CHANNEL_1;
while (ubChannelT < eCOS404_AI_CHANNEL_MAX)
{
//-------------------------------------------------------------------------------------------
// get new ADC value for corresponding channel
//
slFieldValueT = 0x07FF;
//-------------------------------------------------------------------------------------------
// Adjust filter to Field value
//
#if COS_DS404_AI_FILTER > 0
if (aubFilterTypeS[ubChannelT] != 0)
{
}
#endif
//-------------------------------------------------------------------------------------------
// Scale field value and store the result in Process Value
//
slProcessValueT = slFieldValueT * 1 / 1;
//-------------------------------------------------------------------------------------------
//
Cos404_AI_SetStatus(ubChannelT, eCOS404_AI_STATUS_OK);
//-------------------------------------------------------------------------------------------
// pass new data to COS 404 module
//
#if COS_DS404_DT_INT16 == 1
Cos404_AI_SetFV16(ubChannelT, (int16_t)slFieldValueT);
Cos404_AI_SetPV16(ubChannelT, (int16_t)slProcessValueT);
#endif
#if COS_DS404_DT_INT32 == 1
Cos404_AI_SetFV32(ubChannelT, slFieldValueT);
Cos404_AI_SetPV32(ubChannelT, slProcessValueT);
#endif
ubChannelT++;
}
}
//--------------------------------------------------------------------------------------------------------------------//
// Cos404_AI_FilterOrder() //
// //
//--------------------------------------------------------------------------------------------------------------------//
#if COS_DS404_AI_FILTER > 0
uint8_t Cos404_AI_FilterOrder(uint8_t ubChannelV, uint16_t uwOrderV)
{
//---------------------------------------------------------------------------------------------------
// Check the channel parameter first, the channel index starts at eCOS404_AI_CHANNEL_1
//
if (ubChannelV >= eCOS404_AI_CHANNEL_MAX)
{
return eCosSdo_ERR_NO_SUB_INDEX;
}
//---------------------------------------------------------------------------------------------------
//
if ((uwOrderV > 0) && (uwOrderV < 40))
{
auwFilterOrderS[ubChannelV] = uwOrderV;
}
//---------------------------------------------------------------------------------------------------
// value is not valid quit with an error
//
else
{
return (eCosSdo_ERR_VALUE_RANGE);
}
return (eCosSdo_WRITE_OK);
}
#endif

// Cos404_AI_Type() //
// //
//--------------------------------------------------------------------------------------------------------------------//
uint8_t Cos404_AI_Type(uint8_t ubChannelV, uint16_t uwTypeV)
{
//---------------------------------------------------------------------------------------------------
// check the channel parameter first, the channel index starts
// at 0 (first channel)
//
if (ubChannelV >= eCOS404_AI_CHANNEL_MAX)
{
return eCosSdo_ERR_NO_SUB_INDEX;
}
//---------------------------------------------------------------------------------------------------
//\todo check for valid sensor type
//
if (uwTypeV == eCOS404_AI_ST_VOLT)
{
//--------------------------------------------------------
//
//--------------------------------------------------------
// register supported EMCYs of CiA 404 Profile
//
#if COS_DICT_OBJ_1014 > 0
CosEmcyErrorRegister(eEMCY_404_HW_SENSOR_FAULT_CH_1+ubChannelV, EMCY_ERR_HW_SENSOR_FAULT, eCoErrReg_VOLTAGE, 1);
CosEmcyErrorRegister(eEMCY_404_HW_GENERIC, EMCY_ERR_HW_GENERIC, eCoErrReg_MANUFACTURER, 0);
#endif
//--------------------------------------------------------
// set channel description
//
#if COS_DS404_DT_INT16 > 0 // 16 bit value
Cos404_AI_SetPVDescription(ubChannelV, 1, 0x00260000); // V with 1DD
#endif
#if COS_DS404_DT_INT32 > 0 // 32 bit value
Cos404_AI_SetPVDescription(ubChannelV, 3, 0x00260000); // V with 3DD
#endif
}
//---------------------------------------------------------------------------------------------------
// sensor type is not valid quit with an error
//
else
{
return (eCosSdo_ERR_VALUE_RANGE);
}
return (eCosSdo_WRITE_OK);
}
